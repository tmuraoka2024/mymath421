library(ggplot2)

nasa_keyword %>% 
  count(keyword, sort = TRUE) %>% 
  filter(nchar(keyword)>1) %>% 
  head(20) %>% 
  ggplot(aes(x = n, y = reorder(keyword, n))) +
  geom_col(fill='dodgerblue') +
  labs(y = '', x = 'Frequency')
